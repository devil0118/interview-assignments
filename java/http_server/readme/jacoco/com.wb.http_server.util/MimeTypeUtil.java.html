    public static String getTypes(String fileName) {
        for(int i=0;i<MIME_MapTable.length;i++){
            if(fileName.endsWith(MIME_MapTable[i][0]))
                return MIME_MapTable[i][1];
        }
        return DEFAULT_CONTENT_TYPE;
    }
